For Loop

For


  for (counter, limit; condition; counter_increment) {
    code block to be executed
  }

Example



  for (var counter = 0, limit = 5; counter < limit; counter++) {
      console.log("The number is " + counter);
  }

Example


  var animalArray = ["Cat", "Dog", "Fish"];

  for (var counter = 0, limit = animalArray.length; counter < limit; counter++) {
      console.log("The animal is " + animalArray[counter)];
  }

Example


  var animalArray = ["Cat", "Dog", "Fish"];

  for (var i = 0, j = animalArray.length; i < j; i++) {
      console.log("The animal is " + animalArray[i)];
  }
